// Play a chord on guitar: longer sustain, cleaner attack function playChord(chordLetter, time) { const root = rootNoteMap[chordLetter]; if (!root) return; const rootMidi = noteToMidi(root); const intervals = chordMap[chordLetter] || [0, 7]; const notes = intervals.map(i => midiToNoteName(rootMidi + i)); guitar.triggerAttack(notes, time); guitar.triggerRelease(time + Tone.Time("2n").toSeconds()); // hold 2 quarter notes } // Play bass root + octave with similar sustain function playBass(chordLetter, time) { const root = rootNoteMap[chordLetter]; if (!root) return; const rootMidi = noteToMidi(root); const octaveNote = midiToNoteName(rootMidi + 12); bass.triggerAttack([root, octaveNote], time); bass.triggerRelease(time + Tone.Time("2n").toSeconds()); }
